Selecionando e inserindo MySQL

A instrução SELECT INTO em MySQL é usada para guardar os calores retornadados, de uma única linha, em variáveis que podem ser usadas posteriormente para cálculos ou até mesmo inserir dados em outra tabela.SELECT coluna1, coluna2, ... INTO @VariávelColuna1, @VariávelColuna2, ... (IN outro_banco_de_dados) FROM tabela WHERE condição, onde o trecho de instrução (IN outro_banco_de_dados) é opcional e é usado quando a tabela é de outro banco de dados. Lembrando, que a consulta deve ser retornada uma única instrução. Caso necessite, pode usar o operador LIMIT 1.

CREATE TABLE clientes(
  cod_cli INT primary key,
  nome_cli VARCHAR(45),
  rua_cli VARCHAR(45),
  bairro_cli INT,
  num_cli INT,
  fone_cli VARCHAR(9)
);

CREATE TABLE carros(
  placa_car VARCHAR6(45)) primary key,
  cliente_car INT,
  marca_car VARCHAR(45),
  modelo_car VARCHAR(45),
  cor_car VARCHAR(45)
);

ALTER TABLE carros add foreign key (cliente_car) references clientes (cod_cli);

INSERT INTO SELECT

A instrução INSERT INTO SELECT é usada para inserir dados de outra tabela em uma tabela já existente. A sintaxe base é INSERT INTO tabelaA (coluna1, coluna2,...) SELECT (coluna1, coluna2,...) FROM tabelaB WHERE condição;.

CREATE TABLE carro_cliente(
  placa VARCHAR(6),
  fone VARCHAR(9),
  cod INT,
  nome VARCHAR(45)
);


-- Inserindo o resultado da consulta em uma tabela
INSERT INTO carro_cliente (placa, fone, cod, nome) SELECT placa_car, fone_cli, cod_cli, nome_cli FROM clientes INNER JOIN carros ON cod_cli = cliente_car;
