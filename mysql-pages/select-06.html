Funções de Texto com MySQL

Existem diversas funções prontas em MySQL para trabalhar com string. Funções que deixam o textos em caixa baixa ou caixa alta, retornam o tamanho de caracteres do texto, inserem palavras em um texto, concatenam dois textos em um único texto, pesquisam de se existem uma palavra específica no texto, dentro outras várias funções.

CREATE TABLE textos(
  nome VARCHAR(45),
  sobrenome VARCHAR(45)
);

INSERT INTO textos VALUES ('FULANO', 'de tal'),('Sicrano', 'da silva'),('BELTRANO', 'de SOUZA');
SELECT * FROM textos;
-- Concatenando o nome e o sobrenome e um espaço entre eles

nome sobrenome
FULANO de tal
Sicrano da silva
BELTRANO de SOUZA

SELECT concat(nome,' ', sobrenome) FROM  textos;

concat(nome,' ', sobrenome)
FULANO de tal
Sicrano da silva
BELTRANO de SOUZA

-- Coloando o nome em letras minúsculas e o sobrenome em letras maiúsculas
SELECT lower(nome), upper(sobrenome) FROM textos;

nome sobrenome
fulano DE TAL
Sicrano DA SILVA
beltrano DE SOUZA

-- Inserindo 'Sr. no inicio do nome concatenado e retornando o tamanho do nome com espaços'
SELECT insert(concat(nome,' ', sobrenome), 1, 0, 'Sr. '), length(insert(concat(nome,' ', sobrenome), 1, 0, 'Sr. ')) FROM textos;

nome sobrenome
Sr. FULANO de tal 17
Sr. Sicrano da silva 20
Sr. BELTRANO de SOUZA 22
