Atualizando dados com MySQL

Com a instrução UPDATE podemos atualizar os dados de uma tabela que já foram previamente inseridos. A sintaxe base é UPDATE tabela SET coluna = novoValor, coluna2 = novoValor2, ..., WHERE condição.

A cláusula WHERE é essencial, pois ao omití-la iremos atualizar todos os dados da coluna para o novo valor. Usando uma condição, poderemos atualizar somente os dados que a atenderem e não todos (a menos que deseje). A condição mais comum é atualizar os dados pelo valor de sua chave primária, pois é único.

CREATE TABLE produtos(
  id_produto INT primary key auto_increment,
  nome TEXT,
  preco DOUBLE
);
INSERT INTO produtos (id_produto, nome, preco) values (1, 'Teclado', 10.50);
INSERT INTO produtos (nome, preco) values ('Mouse', 9.90);
INSERT INTO produtos values (3, 'Monitor', 99.90);

--Buscando todas as colunas da tabela
SELECT * FROM produtos;

id_produto nome preco
1 Teclado 10.5
2 Mouse 9.9
3 Monitor 99.9

-- Atualizando o nome do produto com ID 3
UPDATE produtos SET nome = 'Tapete de Mouse' WHERE id_produto = 3;

SELECT * FROM produtos;

id_produto nome preco
1 Teclado 10.5
2 Mouse 9.9
3 Tapete de Mouse 99.9

-- Atualizando o nome de todos os produtos com id igual o menor que 2
UPDATE produtos SET nome = 'Ratos', preco = '10.90' WHERE id_produto <= 2;

SELECT * FROM produtos;

id_produto nome preco
1 Ratos 10.9
2 Ratos 10.90
3 Tapete de Mouse 99.9

ON UPDATE RESTRICT - Mantendo a consistência dos dados

A restrição ON UPDATE RESTRICT já é inserida por padrão em chaves estrangeiras, e impede que a chave primária, que seja estrangeira em outra tabela, seja alterada. Basicamente não podemos atualizar a chave primária, a menos que atualizemos todos as chaves estrangeiras ligadas à ela anteriormente.

ON UPDATE CASCADE -Mantendo a consistência dos dados

A restrição ON UPDATE CASCADE é usada quando queremos que, ao atualizar uma chave primária, todas as chaves estrangeiras ligadas àquela chave, também sejam atualizadas automáticamente.

CREATE TABLE pais(
  cod_pais INT primary key auto_increment,
  nome_pais VARCHAR(60) not null
);

CREATE TABLE usuario(
  cod_usuario INT primary key auto_increment,
  nome_user VARCHAR(50) not null,
  cod_pais INT,
  CONSTRAINT FK_USUARIO_PAIS5 FOREIGN KEY (COD_PAIS) REFERENCES PAIS5 (COD_PAIS) ON UPDATE CASCADE
);

INSERT INTO pais (cod_pais, nome_pais) VALUES
(1,'Brasil'),
(2, 'Paraguai'),
(3, 'Chile');

INSERT INTO usuario VALUES
(1,'Pablo', 2),
(2, 'João', 1),
(3,'Maria', 1);

SELECT * FROM PAIS;

cod_pais nome_pais
1 Brasil
2 Paraguai
3 Chile

SELECT * FROM USUARIO;

cod_usuario nome_user cod_pais
1 Pablo 2
2 João 1
3 Maria 1

-- Atualizando a chave primária do país 1 para 10
UPDATE PAIS SET COD_PAIS = 10 WHERE COD_PAIS = 1;

SELECT * FROM PAIS;

cod_pais nome_pais
2 Paraguai
3 Chile
10 Brasil

SELECT * FROM USUARIO;

cod_usuario nome_user cod_pais
1 Pablo 2
2 João 10
3 Maria 10
